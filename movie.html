function InfoUpdate() { if ($("#Name").val() != null && $("#Name").val() != '') { AddToTable(); formClear(); $("#Name").focus(); } if ($("#Id").val() != null && $("#Id").val() != '') { AddToTable(); formClear(); $("#Id").focus(); } if ($("#City").val() != null && $("#City").val() != '') { AddToTable(); formClear(); $("#City").focus(); } if ($("#Degree").val() != null && $("#Degree").val() != '') { AddToTable(); formClear(); $("#Degree").focus(); } if ($("#Timestamp").val() != null && $("#Timestamp").val() != '') { AddToTable(); formClear(); $("#Timestamp").focus(); } } function AddToTable() { // First check if a tag exists, add one if not if ($("#Table tbody").length == 0) { $("#Table").append(""); } $("#Table tbody").append("" + "" + $("#Name").val() + "" + "" + $("#Id").val() + "" + "" + $("#City").val() + "" + "" + $("#Degree").val() + "" + "" + $("#Timestamp").val() + "" + ""); } function formClear() { $("#Name").val(""); $("#Id").val(""); $("#City").val(""); $("#Degree").val(""); $("#Timestamp").val(""); }
